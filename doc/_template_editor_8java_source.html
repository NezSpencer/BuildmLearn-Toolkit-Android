464  dataElement.appendChild(item);
465  }

511 
512  try {

514  DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
515  DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
516  Document doc;
517  doc = dBuilder.parse(fXmlFile);
518  doc.getDocumentElement().normalize();
519  Element authorElement = (Element) doc.getElementsByTagName("author").item(0);
520  Element nameElement = (Element) authorElement.getElementsByTagName("name").item(0);
521  String name = nameElement.getTextContent();
522  String title = doc.getElementsByTagName("title").item(0).getTextContent();
523  NodeList nList = doc.getElementsByTagName("item");
524  ArrayList<Element> items = new ArrayList<>();
525  for (int i = 0; i < nList.getLength(); i++) {
526  Node nodeItem = nList.item(i);
527  if (nodeItem.getNodeType() == Node.ELEMENT_NODE) {
528  items.add((Element) nodeItem);
529  }
